INCLUDE "EMU8086.INC"
.MODEL SMALL
.STACK 100H
.DATA  

MSG1 DW "ENTER ANY CHARACTER ( A-Z ) =  $"
MSG2 DW "YOU HAVE ENTERED = $" 
MSG3 DW "YOU HAVE ENTERED INVALID INPUT $" 

.CODE   
MAIN PROC
    
    MOV AX,@DATA
    MOV DS, AX
    
    MOV BL,80H
    MOV BH,10
    MOV CX,127
    
    MOV AH,2
    
    TOP:
    MOV DL,BL
    INT 21H
    
    INC BL
    DEC CX
    
    JCXZ EXIT
    DEC BH
    JE LEVEL1
    JMP TOP
    
    LEVEL1:
    MOV BH,10
    PRINTN ""
    JMP TOP
    
    
    
    EXIT:   
    MOV AH, 4CH
    INT 21H
    MAIN ENDP
END MAIN
