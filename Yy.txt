INCLUDE "EMU8086.INC"
.MODEL SMALL
.STACK 100H
.DATA  

MSG1 DW "ENTER Y or y =  $"
MSG2 DW "YOU HAVE ENTERED = $" 
MSG3 DW "YOU HAVE INVALID INPUT $" 

.CODE   
MAIN PROC
    MOV AX,@DATA
    MOV DS,AX
    
    LEA DX,MSG1
    MOV AH,9
    INT 21H
    
    MOV AH,1
    INT 21H
    MOV BL,AL 
    
    PRINTN ""
    
    CMP BL, 'Y'
    JE LEVEL1
    CMP BL,'y'
    JE LEVEL1 
    
    
    LEA DX,MSG3
    MOV AH,9
    INT 21H 
    JMP EXIT
    
     
    
    
    LEVEL1:    
    LEA DX,MSG2
    MOV AH,9
    INT 21H 
    MOV AH,2
    MOV DL,BL
    INT 21H
    JMP EXIT
    
    
    
    
    
    EXIT:   
    MOV AH, 4CH
    INT 21H
    MAIN ENDP
END MAIN
