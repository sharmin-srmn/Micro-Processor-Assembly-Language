
INCLUDE "EMU8086.INC"
.MODEL SMALL
.STACK 100H
.DATA 

MSG1 DB "ENTER FIRST NUMBER = $"
MSG2 DB "ENTER SECOND NUMBER = $"
MSG3 DB "THE ADDITION IS = $"
 
.CODE   

MAIN PROC 
    
    MOV AX,@DATA    ; INITIALIZE DATA
    MOV DS,AX
    
    LEA DX,MSG1     ; FIRST MESSAGE
    MOV AH,9
    INT 21H      
    
    MOV AH,1
    INT 21H    
    MOV BL,AL       ; TAKING INPUT OF FIRST NUMBER
    INT 21H    
    MOV BH,AL 
    
    PRINTN ""       ; PRINT NEW LINE
    
    
    LEA DX,MSG2
    MOV AH,9        ; SECOND NUMBER MESSAGE
    INT 21H
    
    MOV AH,1
    INT 21H         ; TAKING INPUT OF SECOND NUMBER
    MOV CL,AL
    INT 21H
    MOV CH,AL
    
    ADD BH,CH
    SUB BH,48
    
    ADD BL,CL
    SUB BL,48       
    
    
    MOV AH,2
    MOV DL,0DH
    INT 21H         ; NEW LINE 
    MOV DL,0AH
    INT 21H
    
    LEA DX,MSG3
    MOV AH,9       ; THIRD MESSAGE
    INT 21H
    
    MOV AH,2    
    MOV DL,BL
    INT 21H 
    MOV DL,BH      ; FIRST NUMBER OUTPUT
    INT 21H  
    
    PRINTN ""      ; NEW LINE
   
    
    
    
    
    
    
    
    
      
      MOV AH,4CH
      INT 21H
    
    
    MAIN ENDP
END MAIN
